"""
Export utilities for generating downloadable reports and data files.
"""

import io
import pandas as pd
from typing import Dict, List, Optional


def export_risk_scores_csv(risk_scores_df: pd.DataFrame) -> bytes:
    """Export risk scores table as CSV."""
    export_cols = [
        "rank", "country", "overall_score", "rating",
        "structural_score", "anomaly_score", "mc_leakage_score", "governance_score"
    ]
    available = [c for c in export_cols if c in risk_scores_df.columns]
    return risk_scores_df[available].to_csv(index=False).encode("utf-8")


def export_anomaly_data_csv(anomaly_df: pd.DataFrame) -> bytes:
    """Export anomaly detection results as CSV."""
    export_cols = [
        "reporter", "partner", "hs_category", "hs_description", "year",
        "export_value_usd", "import_value_usd",
        "composite_anomaly_score", "risk_level",
        "spike_flag", "spike_zscore",
        "capacity_mismatch_flag", "capacity_ratio",
        "origin_shift_flag", "origin_shift_score",
    ]
    available = [c for c in export_cols if c in anomaly_df.columns]
    return anomaly_df[available].to_csv(index=False).encode("utf-8")


def export_simulation_summary_csv(mc_results: Dict[str, Dict]) -> bytes:
    """Export Monte Carlo simulation summary as CSV."""
    rows = []
    for country, results in mc_results.items():
        rows.append({
            "country": country,
            "scenario": results.get("scenario_name", "N/A"),
            "final_leakage_mean_pct": round(results["final_leakage_mean"] * 100, 2),
            "final_leakage_ci_low_pct": round(results["final_leakage_ci_90"][0] * 100, 2),
            "final_leakage_ci_high_pct": round(results["final_leakage_ci_90"][1] * 100, 2),
            "final_circumvention_mean_pct": round(results["final_circumvention_mean"] * 100, 2),
            "n_simulations": results["n_simulations"],
        })
    return pd.DataFrame(rows).to_csv(index=False).encode("utf-8")


def export_policy_recommendations_csv(recommendations: List[Dict]) -> bytes:
    """Export policy recommendations as CSV."""
    return pd.DataFrame(recommendations).to_csv(index=False).encode("utf-8")


def generate_executive_summary_text(
    risk_scores_df: pd.DataFrame,
    mc_results: Optional[Dict] = None,
    scenario_name: str = "Baseline"
) -> str:
    """Generate plain-text executive summary for download."""
    n_total = len(risk_scores_df)
    n_critical = len(risk_scores_df[risk_scores_df["rating"] == "Critical"])
    n_high = len(risk_scores_df[risk_scores_df["rating"] == "High"])
    
    top3 = risk_scores_df.head(3)
    
    text = f"""
================================================================================
   AFRICAN EPA-AfCFTA OVERLAP INTEGRITY PLATFORM
   Executive Risk Summary
   Scenario: {scenario_name}
================================================================================

OVERVIEW
--------
Countries assessed: {n_total}
Risk distribution:
  - Critical: {n_critical}
  - High: {n_high}
  - Moderate: {len(risk_scores_df[risk_scores_df["rating"] == "Moderate"])}
  - Low: {len(risk_scores_df[risk_scores_df["rating"] == "Low"])}

TOP-3 RISK-RATED COUNTRIES
---------------------------
"""
    for i, (_, row) in enumerate(top3.iterrows(), 1):
        text += f"""
{i}. {row['country']} — Score: {row['overall_score']:.1f}/100 ({row['rating']})
   Structural: {row['structural_score']:.1f}  |  Anomaly: {row['anomaly_score']:.1f}  |  
   MC Leakage: {row['mc_leakage_score']:.1f}  |  Governance: {row['governance_score']:.1f}
"""
    
    if mc_results:
        text += """
MONTE CARLO SIMULATION RESULTS
-------------------------------
"""
        for country, results in mc_results.items():
            text += f"""
{country}:
  Expected leakage: {results['final_leakage_mean']*100:.1f}% 
  (90% CI: {results['final_leakage_ci_90'][0]*100:.1f}% - {results['final_leakage_ci_90'][1]*100:.1f}%)
  Circumvention rate: {results['final_circumvention_mean']*100:.1f}%
"""
    
    text += """
================================================================================
DISCLAIMER: This analysis uses simulation-based modeling with synthetic data 
calibrated to public sources. Results should be interpreted as indicative risk 
assessments, not precise predictions. Framed as neutral capacity-building input.
================================================================================
"""
    
    return text
